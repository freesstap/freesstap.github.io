常用的时间转换

元组时间转换为时间戳

import time tuple_time=time.localtime()print(tuple_time)#转换为时间戳print(time.mktime(tuple_time))

2\元组时间转换为字符串时间
import time
tuple_time=time.localtime()
print(tuple_time)


3\时间戳类型的时间转换为字符串时间
import os
import time
pwd_time=os.path.getatime(’/etc/passwd’)
print(pwd_time)
print(time.ctime(pwd_time))

4\时间戳转换为元组
